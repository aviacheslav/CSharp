            if(!string.IsNullOrEmpty(textTrainN.Text))nomerPoezda = Convert.ToInt32(textTrainN.Text);
            from = txtFrom.Text;
            to = txtTo.Text;

            SqlConnectionStringBuilder builder=new SqlConnectionStringBuilder();
            try
            {
                builder.DataSource = "10.3.0.5";
                builder.InitialCatalog = "06747_zdtickets";
                builder.IntegratedSecurity = false;
                builder.UserID = "pp2022";
                builder.Password = "pp2022";
                connection = new SqlConnection();
                connection.ConnectionString = builder.ConnectionString;
                
                connection.Open();

                string select = "Select  train_id as 'Номер поезда', city_from_name as 'Город отправления',city_to_name as 'Город прибытия',rout_date as 'Дата отправления',countPK as 'Кол-во плацкартных мест',pricePK as 'Цена плацкартных мест',countSV as 'Кол-во мест СВ',priceSV as 'Цена места СВ',countKupe as 'Кол-во мест купе',priceKupe as 'Цена места купе' from v_trains";

                    }
